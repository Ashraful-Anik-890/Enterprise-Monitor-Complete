
def get_running_apps_with_pids() -> List[Dict[str, List[str]]]:
    """
    Returns list of running GUI apps with names and PIDs using a reliable JSON-based approach
    """
    script = """
        on convert_to_json(theList)
            set jsonList to ""
            repeat with itemRef in theList
                set {appName, appPID, windowNames} to itemRef
                set windowNamesJson to "["
                repeat with windowName in windowNames
                    set windowNamesJson to windowNamesJson & "\\"" & windowName & "\\","
                end repeat
                if length of windowNames > 0 then
                    set windowNamesJson to text 1 thru -2 of windowNamesJson
                end if
                set windowNamesJson to windowNamesJson & "]"

                set jsonList to jsonList & "{\\"name\\":\\"" & appName & "\\",\\"pid\\":" & appPID & ",\\"windows\\":" & windowNamesJson & "},"
            end repeat
            return "[" & text 1 thru -2 of jsonList & "]"
        end convert_to_json

        tell application "System Events"
            set results to {}
            set processList to every process whose visible is true

            repeat with proc in processList
                try
                    set appName to name of proc
                    set appPID to unix id of proc
                    set windowNames to {}

                    try
                        tell proc
                            set windowNames to name of every window
                        end tell
                    on error errMsg
                        log "Window error for " & appName & ": " & errMsg
                    end try

                    set end of results to {appName, appPID, windowNames}
                on error errMsg
                    log "Process error: " & errMsg
                end try
            end repeat
        end tell

        return convert_to_json(results)
    """
    try:
        result = subprocess.run(
            ['osascript', '-e', script],
            capture_output=True,
            text=True
        )
        print("Raw Output:", result.stdout)
        print("Errors:", result.stderr)  # Debug logs appear here!
        return json.loads(result.stdout)

    except subprocess.CalledProcessError as e:
        print(f"AppleScript Error: {e.stderr.strip()}")
        return []
    except subprocess.SubprocessError as e:
        print(f"Error running AppleScript: {e}")
        return []
    except json.JSONDecodeError:
        print("Failed to parse AppleScript output")
        return []
    except Exception as e:
        print(f"Unexpected error: {str(e)}")
        return []

def get_frontmost_window_title():
    script = """
	tell application "System Events"
		try
			set frontApp to first process whose frontmost is true
			return {name of frontApp, id of frontApp}
		on error
			return {"Unknown", -1}
		end try
	end tell
    """
    try:
        result = subprocess.run(['osascript', '-e', script],
                                capture_output=True,

                                text=True)
        return result.stdout.strip()
    except subprocess.SubprocessError as e:
        print(f"Error getting window title: {e}")
        return None
